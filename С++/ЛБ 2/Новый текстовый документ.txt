#include "stdafx.h"
#include "iostream"
using namespace std;
char* f(char* s, char** e) {
	char* p, *i, *j;
	while (*s) {
		p = s + 1;
		while (*p) {
			if (*p == *s && *(p + 1) == *(s + 1))
				break;
			p++;
		}
		if ((p - s) > 1) {
			i = s;
			j = p;
			while (*i && (*i == *j)) {
				j++;
				i++;
			}

			if (i == p) {
				if (e != 0)
					*e = j;
				return s;
			}
		}
		s++;
	}
	return 0;
}
